Covid 19

Covid 19 data

get data

owid_covid_data <- read_csv(here::here('content/projects/project2', 'owid-covid-data.csv'))

total cases and deaths by continent

create a new data set by continent and date

covid_region <- owid_covid_data %>% 
  filter(continent!="NA") %>% 
  group_by(continent, date) %>% 
  summarise(total_cases_cotinent = sum(total_cases, na.rm = TRUE),
            total_deaths_continent = sum(total_deaths, na.rm = TRUE))

total cases by continent

covid_region %>% 
  ggplot(aes(x = date, y = total_cases_cotinent, fill = continent)) +
  geom_area() +
  theme_bw()
